```javascript
/**
 * Script de b√∫squeda AJAX para Glory Framework
 */
class BusquedaAjax {
    constructor() {
        this.inputs = document.querySelectorAll('.busqueda-input[data-ajax-url]');
        this.debounceTimer = null;
        this.minQueryLength = 2;
        this.debounceDelay = 300;

        this.init();
    }

    init() {
        this.inputs.forEach(input => {
            input.addEventListener('input', (e) => this.onInput(e));
            input.addEventListener('focus', (e) => this.onFocus(e));
            input.addEventListener('blur', (e) => this.onBlur(e));
        });

        // Cerrar resultados al hacer click fuera
        document.addEventListener('click', (e) => this.onDocumentClick(e));
    }

    onInput(e) {
        const input = e.target;
        const query = input.value.trim();

        clearTimeout(this.debounceTimer);

        if (query.length >= this.minQueryLength) {
            this.debounceTimer = setTimeout(() => {
                this.buscar(query, input);
            }, this.debounceDelay);
        } else {
            this.ocultarResultados(input);
        }
    }

    onFocus(e) {
        const input = e.target;
        const resultsContainer = this.getResultsContainer(input);

        if (resultsContainer && resultsContainer.innerHTML.trim()) {
            resultsContainer.style.display = 'block';
        }
    }

    onBlur(e) {
        // Delay para permitir clicks en resultados
        setTimeout(() => {
            const input = e.target;
            const resultsContainer = this.getResultsContainer(input);

            if (resultsContainer && !resultsContainer.matches(':hover')) {
                resultsContainer.style.display = 'none';
            }
        }, 150);
    }

    onDocumentClick(e) {
        if (!e.target.closest('.busqueda-contenedor')) {
            document.querySelectorAll('.resultados-busqueda').forEach(container => {
                container.style.display = 'none';
            });
        }
    }

    async buscar(query, input) {
        const ajaxUrl = input.dataset.ajaxUrl;
        const action = input.dataset.action;
        const resultsContainer = this.getResultsContainer(input);

        if (!ajaxUrl || !action || !resultsContainer) {
            console.error('Configuraci√≥n AJAX incompleta para b√∫squeda');
            return;
        }

        try {
            // Mostrar loading
            resultsContainer.innerHTML = '<div class="busqueda-loading">Buscando...</div>';
            resultsContainer.style.display = 'block';

            const response = await fetch(ajaxUrl, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: new URLSearchParams({
                    action: action,
                    query: query,
                    security: gloryAjax?.nonce || ''
                })
            });

            const data = await response.json();

            if (data.success) {
                resultsContainer.innerHTML = data.data.html;

                // Mostrar resultados
                if (data.data.html.trim()) {
                    resultsContainer.style.display = 'block';
                } else {
                    this.ocultarResultados(input);
                }
            } else {
                throw new Error(data.data || 'Error en b√∫squeda');
            }

        } catch (error) {
            console.error('Error en b√∫squeda AJAX:', error);
            resultsContainer.innerHTML = '<div class="busqueda-error">Error al buscar. Intenta nuevamente.</div>';
        }
    }

    ocultarResultados(input) {
        const resultsContainer = this.getResultsContainer(input);
        if (resultsContainer) {
            resultsContainer.style.display = 'none';
            resultsContainer.innerHTML = '';
        }
    }

    getResultsContainer(input) {
        const container = input.closest('.busqueda-contenedor');
        return container ? container.querySelector('.resultados-busqueda') : null;
    }
}

// Inicializar cuando DOM est√© listo
document.addEventListener('DOMContentLoaded', () => {
    new BusquedaAjax();
});

// Tambi√©n inicializar cuando GloryAjax est√© disponible
if (typeof gloryAjax !== 'undefined') {
    gloryAjax.ready(() => {
        new BusquedaAjax();
    });
}
```

```php
<?php
/**
 * Handlers AJAX para b√∫squeda en Glory Framework
 */

class AjaxSearchHandler {
    public static function init() {
        add_action('wp_ajax_busqueda_ajax', [__CLASS__, 'handleAjaxSearch']);
        add_action('wp_ajax_nopriv_busqueda_ajax', [__CLASS__, 'handleAjaxSearch']);
    }

    public static function handleAjaxSearch() {
        try {
            // Verificar nonce si est√° disponible
            if (isset($_POST['security']) && !wp_verify_nonce($_POST['security'], 'glory_ajax_nonce')) {
                wp_send_json_error('Security check failed');
                return;
            }

            $query = sanitize_text_field($_POST['query'] ?? '');

            if (empty($query) || strlen($query) < 2) {
                wp_send_json_error('Query too short');
                return;
            }

            // Realizar b√∫squeda
            $resultados = self::performSearch($query);

            // Renderizar con BusquedaRenderer
            use Glory\Components\BusquedaRenderer;
            $html = BusquedaRenderer::renderizarResultados($resultados);

            wp_send_json_success([
                'html' => $html,
                'total' => array_sum(array_map('count', $resultados)),
                'query' => $query
            ]);

        } catch (Exception $e) {
            wp_send_json_error('Search failed: ' . $e->getMessage());
        }
    }

    private static function performSearch(string $query): array {
        $resultados = [];

        // Buscar posts
        $posts = get_posts([
            's' => $query,
            'post_type' => 'post',
            'posts_per_page' => 5,
            'post_status' => 'publish'
        ]);

        foreach ($posts as $post) {
            $resultados['posts'][] = [
                'titulo' => get_the_title($post),
                'url' => get_permalink($post),
                'tipo' => 'Post',
                'imagen' => get_the_post_thumbnail_url($post, 'thumbnail')
            ];
        }

        // Buscar p√°ginas
        $pages = get_posts([
            's' => $query,
            'post_type' => 'page',
            'posts_per_page' => 3,
            'post_status' => 'publish'
        ]);

        foreach ($pages as $page) {
            $resultados['pages'][] = [
                'titulo' => get_the_title($page),
                'url' => get_permalink($page),
                'tipo' => 'P√°gina',
                'imagen' => get_the_post_thumbnail_url($page, 'thumbnail')
            ];
        }

        return $resultados;
    }
}
```

```php
    {
        $defaults = [
            'query' => '',
            'tipos' => self::$tiposSoportados,
            'limite' => 20,
            'pagina' => 1
        ];

        $config = array_merge($defaults, $args);

        // Realizar b√∫squeda
        $resultados = self::buscarContenido($config);

        // Renderizar header de resultados

    private static function buscarContenido(array $config): array
    {
        $resultados = [];

            $posts = get_posts([
                's' => $config['query'],
                'post_type' => $tipo,
                'posts_per_page' => $config['limite'],
                'paged' => $config['pagina']
            ]);

                $resultados[$tipo . 's'][] = [
                    'titulo' => get_the_title($post),
                    'url' => get_permalink($post),
                    'tipo' => ucfirst($tipo),
                    'imagen' => get_the_post_thumbnail_url($post, 'thumbnail')
                ];
            }
        }

        return $resultados;
    }

        } else {
            echo "<p>No se encontraron resultados para \"{$config['query']}\"</p>";
        }

        echo '</div>';
    }
```

### B√∫squeda con Filtros

```php
<?php
use Glory\Components\BusquedaRenderer;

class BuscadorConFiltros
{
    public static function renderizarBusquedaFiltrada(): void
    {
        $query = $_GET['q'] ?? '';
        $tipo = $_GET['tipo'] ?? 'todos';
        $categoria = $_GET['categoria'] ?? '';

        // Obtener resultados filtrados
        $resultados = self::buscarConFiltros($query, $tipo, $categoria);

        // Renderizar filtros
        self::renderizarFiltros($tipo, $categoria);

        // Renderizar resultados
        echo '<div class="resultados-busqueda">';
        echo BusquedaRenderer::renderizarResultados($resultados);
        echo '</div>';
    }

    private static function buscarConFiltros(string $query, string $tipo, string $categoria): array
    {
        $args = [
            's' => $query,
            'posts_per_page' => 20
        ];

        // Filtrar por tipo
        if ($tipo !== 'todos') {
            $args['post_type'] = $tipo;
        } else {
            $args['post_type'] = ['post', 'page'];
        }

        // Filtrar por categor√≠a
        if (!empty($categoria)) {
            $args['category_name'] = $categoria;
        }

        $posts = get_posts($args);
        $resultados = [];

        foreach ($posts as $post) {
            $tipoLabel = $post->post_type === 'post' ? 'Post' : 'P√°gina';

            $resultados[$post->post_type . 's'][] = [
                'titulo' => get_the_title($post),
                'url' => get_permalink($post),
                'tipo' => $tipoLabel,
                'imagen' => get_the_post_thumbnail_url($post, 'thumbnail')
            ];
        }

        return $resultados;
    }

    private static function renderizarFiltros(string $tipoActual, string $categoriaActual): void
    {
        ?>
        <form method="GET" class="filtros-busqueda">
            <input type="text" name="q" value="<?php echo esc_attr($_GET['q'] ?? ''); ?>" placeholder="Buscar...">

                <option value="todos" <?php selected($tipoActual, 'todos'); ?>>Todos los tipos</option>
                <option value="post" <?php selected($tipoActual, 'post'); ?>>Posts</option>
                <option value="page" <?php selected($tipoActual, 'page'); ?>>P√°ginas</option>
            </select>

            <input type="text" name="categoria" value="<?php echo esc_attr($categoriaActual); ?>" placeholder="Categor√≠a">

            <button type="submit">Buscar</button>
        </form>
        <?php
    }
}
?>
```
